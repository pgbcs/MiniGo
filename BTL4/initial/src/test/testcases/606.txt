
        type Index struct {
            value int
        }

        func main() {
            var a[5] int = [5]int{1, 2, 3, 4, 5}
            var b Index = Index{value: 2}
            putInt(a[b.getValue(1)])  
        }
        func (i Index) getValue(off int) int {
            return i.value + off
        }
